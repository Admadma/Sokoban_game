1   package game.model.xmlhandler;

3   import game.model.Entity;
4   import jakarta.xml.bind.JAXBContext;
5   import jakarta.xml.bind.Marshaller;
6   
7   import java.io.FileOutputStream;
8   
9   public class JavaToXml {
10  
11      public static void createXml(int moves, Entity player, Entity[] walls, Entity[] balls, Entity[] goals) throws Exception{
12          var gameState = new GameState();
13  
14          gameState.setNumberOfMoves(moves);
15  
16          int[] playerX = new int[]{player.getPosition().col()};
17          gameState.setPlayerXValues(playerX);
18          int[] playerY = new int[]{player.getPosition().row()};
19          gameState.setPlayerYValues(playerY);
20  
21          int[] wallX = new int[walls.length];
22          for(int i = 0; i < walls.length; i++){
23              wallX[i] = walls[i].getPosition().col();
24          }
25          gameState.setWallXValues(wallX);
26          int[] wallY = new int[walls.length];
27          for(int i = 0; i < walls.length; i++){
28              wallY[i] = walls[i].getPosition().row();
29          }
30          gameState.setWallYValues(wallY);
31  
32          int[] ballX = new int[balls.length];
33          for(int i = 0; i < balls.length; i++){
34              ballX[i] = balls[i].getPosition().col();
35          }
36          gameState.setBallXValues(ballX);
37          int[] ballY = new int[balls.length];
38          for(int i = 0; i < balls.length; i++){
39              ballY[i] = balls[i].getPosition().row();
40          }
41          gameState.setBallYValues(ballY);
42  
43          int[] goalX = new int[goals.length];
44          for(int i = 0; i < goals.length; i++){
45              goalX[i] = goals[i].getPosition().col();
46          }
47          gameState.setGoalXValues(goalX);
48          int[] goalY = new int[goals.length];
49          for(int i = 0; i < goals.length; i++){
50              goalY[i] = goals[i].getPosition().row();
51          }
52          gameState.setGoalYValues(goalY);
53  
54          JAXBContext jaxbContext = JAXBContext.newInstance(GameState.class);
55          Marshaller marshaller = jaxbContext.createMarshaller();
56          marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
57          marshaller.setProperty(Marshaller.JAXB_ENCODING, "UTF-8");
58  
59          String savedDataPath = System.getProperty("user.dir") + "/helperFolder" + "/SavedGameState.xml";
60          marshaller.marshal(gameState, new FileOutputStream(savedDataPath));
61      }
62  }
